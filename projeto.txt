    unsigned int id;
    char marca[20];
    char modelo[20];

    MAQUINA pc;
    struct node *left;
    struct node *right;
    struct node *father;
}NODE;

        maketree(arv, v);
    else if (v<info(*arv)) /* if (v<(*arv)->info) */
